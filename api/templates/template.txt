
#include "ofMain.h"
#include "cpptest.h"

// include 

using namespace std;

// what test 
class {WHAT}Suite 
    : public Test::Suite {
public:
    
    {WHAT}Suite() {
        TEST_ADD( {WHAT}Suite::test_ )
    }
    
protected:
    
    void 
    setup() {}
    
    void 
    tear_down() {}
    
private:
    
    void 
    test_() {
        TEST_ASSERT( 1 == 1 )
    }
    
};

int
main() {
    Test::Suite ts_types;

    ts_types.add( auto_ptr<Test::Suite>( new {WHAT}Suite ) );

    Test::TextOutput output( Test::TextOutput::Verbose );
    
    return ts_types.run( output );
}
